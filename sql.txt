CREATE DATABASE pruebav3;
USE pruebav3;

CREATE TABLE user(
id INT AUTO_INCREMENT,
email varchar(30),
pass varchar(15),
primary key(id)
);

CREATE TABLE fav(
idProductFav int primary key,
name varchar(30),
category varchar(10),
idUser int references user(id)
);

CREATE TABLE cart(
idProductCart int primary key,
name varchar(30),
category varchar(10),
amount int,
idUser int references user(id)
);

alter table fav add foreign key(idUser) references user(id);
alter table cart add foreign key(idUser) references user(id);
